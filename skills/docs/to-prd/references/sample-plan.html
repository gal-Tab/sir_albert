Threaded Comments on Task Cards

Users can add comments (with one level of nested replies) to any task card, with real-time delivery via WebSocket and per-user unread tracking.

New Comment and CommentRead Prisma models store comments and read receipts. A tRPC router exposes CRUD + list endpoints behind the task_comments_v1 feature flag. The client subscribes to a tRPC WebSocket subscription for live comment events. React Query handles cache invalidation on mutation and on incoming WS events. Unread counts are computed server-side via a left-join on CommentRead.

model Comment {
  id        String   @id @default(cuid())
  body      String   @db.Text
  taskId    String
  authorId  String
  parentId  String?  // null = top-level, set = reply
  createdAt DateTime @default(now())

  task      Task     @relation(fields: [taskId], references: [id], onDelete: Cascade)
  author    User     @relation(fields: [authorId], references: [id])
  parent    Comment? @relation("CommentThread", fields: [parentId], references: [id], onDelete: Cascade)
  replies   Comment[] @relation("CommentThread")
  reads     CommentRead[]

  @@index([taskId, createdAt])
  @@index([parentId])
}

model CommentRead {
  id        String   @id @default(cuid())
  commentId String
  userId    String
  readAt    DateTime @default(now())

  comment   Comment  @relation(fields: [commentId], references: [id], onDelete: Cascade)
  user      User     @relation(fields: [userId], references: [id])

  @@unique([commentId, userId])
  @@index([userId])
}

import { t } from '../trpc';
import { isFeatureEnabled } from '@/lib/feature-flags';

export const requireFlag = (flag: string) =>
  t.middleware(async ({ ctx, next }) => {
    if (!await isFeatureEnabled(flag, ctx.session.user.id)) {
      throw new TRPCError({ code: 'NOT_FOUND' });
    }
    return next();
  });

Step 3: Implement the comment router
// src/server/trpc/routers/comment.ts
import { z } from 'zod';
import { router, protectedProcedure } from '../trpc';
import { requireFlag } from '../middleware/featureFlag';
import { FLAGS } from '@/lib/feature-flags';
import { TRPCError } from '@trpc/server';

const flaggedProcedure = protectedProcedure.use(requireFlag(FLAGS.TASK_COMMENTS_V1));

export const commentRouter = router({
  create: flaggedProcedure
    .input(z.object({
      taskId: z.string(),
      body: z.string().min(1).max(5000),
      parentId: z.string().optional(),
    }))
    .mutation(async ({ ctx, input }) => {
      // Enforce max 1 level of nesting
      if (input.parentId) {
        const parent = await ctx.prisma.comment.findUnique({
          where: { id: input.parentId },
          select: { parentId: true },
        });
        if (!parent) throw new TRPCError({ code: 'NOT_FOUND' });
        if (parent.parentId) throw new TRPCError({
          code: 'BAD_REQUEST',
          message: 'Replies cannot be nested more than one level',
        });
      }
      return ctx.prisma.comment.create({
        data: {
          body: input.body,
          taskId: input.taskId,
          authorId: ctx.session.user.id,
          parentId: input.parentId ?? null,
        },
        include: { author: { select: { id: true, name: true, image: true } } },
      });
    }),

  list: flaggedProcedure
    .input(z.object({ taskId: z.string() }))
    .query(async ({ ctx, input }) => {
      return ctx.prisma.comment.findMany({
        where: { taskId: input.taskId, parentId: null },
        orderBy: { createdAt: 'asc' },
        include: {
          author: { select: { id: true, name: true, image: true } },
          replies: {
            orderBy: { createdAt: 'asc' },
            include: { author: { select: { id: true, name: true, image: true } } },
          },
          _count: { select: { reads: { where: { userId: ctx.session.user.id } } } },
        },
      });
    }),
});

// Add to commentRouter in src/server/trpc/routers/comment.ts

unreadCount: flaggedProcedure
  .input(z.object({ taskId: z.string() }))
  .query(async ({ ctx, input }) => {
    const total = await ctx.prisma.comment.count({
      where: { taskId: input.taskId },
    });
    const read = await ctx.prisma.commentRead.count({
      where: {
        userId: ctx.session.user.id,
        comment: { taskId: input.taskId },
      },
    });
    return total - read;
  }),

markAsRead: flaggedProcedure
  .input(z.object({ taskId: z.string() }))
  .mutation(async ({ ctx, input }) => {
    const comments = await ctx.prisma.comment.findMany({
      where: { taskId: input.taskId },
      select: { id: true },
    });
    const data = comments.map((c) => ({
      commentId: c.id,
      userId: ctx.session.user.id,
    }));
    await ctx.prisma.commentRead.createMany({
      data,
      skipDuplicates: true,
    });
    return { success: true };
  }),

// src/components/comments/CommentInput.tsx
'use client';
import { useState } from 'react';
import { trpc } from '@/lib/trpc';

export function CommentInput({ taskId, parentId, onSubmit }: {
  taskId: string;
  parentId?: string;
  onSubmit?: () => void;
}) {
  const [body, setBody] = useState('');
  const utils = trpc.useUtils();
  const create = trpc.comment.create.useMutation({
    onSuccess: () => {
      setBody('');
      utils.comment.list.invalidate({ taskId });
      utils.comment.unreadCount.invalidate({ taskId });
      onSubmit?.();
    },
  });

  return (
    <form onSubmit={(e) => {
      e.preventDefault();
      if (!body.trim()) return;
      create.mutate({ taskId, body: body.trim(), parentId });
    }}>
      <textarea
        value={body}
        onChange={(e) => setBody(e.target.value)}
        placeholder={parentId ? "Write a reply..." : "Add a comment..."}
        rows={2}
      />
      <button type="submit" disabled={!body.trim() || create.isPending}>
        {create.isPending ? 'Sending...' : 'Send'}
      </button>
    </form>
  );
}

Step 3: Create CommentPanel — orchestrates list, subscription, and unread tracking
// src/components/comments/CommentPanel.tsx
'use client';
import { useEffect } from 'react';
import { trpc } from '@/lib/trpc';
import { CommentThread } from './CommentThread';
import { CommentInput } from './CommentInput';

export function CommentPanel({ taskId }: { taskId: string }) {
  const utils = trpc.useUtils();
  const { data: comments, isLoading } = trpc.comment.list.useQuery({ taskId });
  const markAsRead = trpc.comment.markAsRead.useMutation({
    onSuccess: () => utils.comment.unreadCount.invalidate({ taskId }),
  });

  // Subscribe to live updates
  trpc.comment.onNew.useSubscription({ taskId }, {
    onData: () => {
      utils.comment.list.invalidate({ taskId });
      utils.comment.unreadCount.invalidate({ taskId });
    },
  });

  // Mark as read when panel opens
  useEffect(() => {
    markAsRead.mutate({ taskId });
  }, [taskId]);

  if (isLoading) return <div>Loading comments...</div>;

  return (
    <div className="comment-panel">
      {comments?.map((comment) => (
        <CommentThread key={comment.id} comment={comment} taskId={taskId} />
      ))}
      <CommentInput taskId={taskId} />
    </div>
  );
}
